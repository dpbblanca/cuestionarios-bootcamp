<script>
  const p = new URLSearchParams(location.search)
  const id = p.get("q")

  if(!id || !window.QUIZZES || !window.QUIZZES[id]){
    alert("Cuestionario no encontrado")
  }

  const QUIZ = window.QUIZZES[id]
  document.getElementById("title").textContent = QUIZ.meta.title
  document.getElementById("subtitle").textContent = QUIZ.meta.subtitle

  let i=0, puntos=0, aciertos=0

  function mostrar(){
    document.getElementById("counter").textContent = 
      "Pregunta "+(i+1)+" / "+QUIZ.items.length

    const q = QUIZ.items[i]
    document.getElementById("qtext").textContent = q.q

    const box = document.getElementById("options")
    box.innerHTML = ""

    q.o.forEach((t,idx)=>{
      const b = document.createElement("button")
      b.className="opt"
      b.textContent = t
      b.onclick = ()=> responder(idx)
      box.appendChild(b)
    })
  }

  function responder(idx){
    const q = QUIZ.items[i]
    if(idx === q.a){ puntos+=100; aciertos++ }
    i++
    if(i >= QUIZ.items.length){
      document.getElementById("quiz").style.display="none"
      document.getElementById("result").style.display="block"
      document.getElementById("r_points").textContent = puntos
      document.getElementById("r_hits").textContent = aciertos
      document.getElementById("r_total").textContent = QUIZ.items.length
      return
    }
    mostrar()
  }

  mostrar()
</script>
